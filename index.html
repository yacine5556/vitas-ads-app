# database/models.py - مثال باستخدام SQLAlchemy class User(Base): __tablename__ = 'users' id = Column(Integer, primary_key=True) telegram_id = Column(BigInteger, unique=True) balance = Column(Float, default=0.0) referral_code = Column(String, unique=True) referred_by = Column(BigInteger, ForeignKey('users.telegram_id')) vip_level = Column(String, default='bronze') last_vip_task = Column(DateTime) class Transaction(Base): __tablename__ = 'transactions' id = Column(Integer, primary_key=True) user_id = Column(BigInteger, ForeignKey('users.telegram_id')) amount = Column(Float) type = Column(String) # 'ad_reward', 'referral_bonus', 'vip_task' timestamp = Column(DateTime, default=datetime.utcnow)
